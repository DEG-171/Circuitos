library ieee;
use ieee.std_logic_1164.all;

entity mux4x1_REG_IN_8 is
generic(N : natural:= 32);
port( MAP1_8,MAP2_8,MAP3_8,MAP4_8: in std_logic_vector((N-1) downto 0);
		SW: in std_logic_vector(8 downto 7);
		REG_IN_8: out std_logic_vector((N-1) downto 0)
		);
end mux4x1_REG_IN_8;

architecture mux of mux4x1_REG_IN_8 is
begin
REG_IN_8 <= MAP1_8 when SW = "00" else
	MAP2_8 when SW = "01" else
	MAP3_8 when SW = "10" else
	MAP4_8;
end mux;
